                    <div class="flex gap-2">

                        <select onchange="updateAjuste(${a.id}, 'signo', this.value)" class="text-sm sm:text-base border-gray-300 rounded px-2 py-2 bg-gray-50 outline-none border">

                            <option value="1" ${a.signo == 1 ? 'selected' : ''}>Suma (+)</option>

                            <option value="-1" ${a.signo == -1 ? 'selected' : ''}>Resta (-)</option>

                        </div>

                    </div>

                    <div class="flex gap-2 items-center">

                        <input type="text" placeholder="Motivo del ajuste (Texto)" value="${a.motivo}" oninput="updateAjuste(${a.id}, 'motivo', this.value)" class="w-full text-sm sm:text-base border-gray-300 rounded px-3 py-2 bg-gray-50 focus:ring-2 focus:ring-purple-500 outline-none border">

                </div>

        const denoms = [

            {v: 20, l: 'Moneda'}, {v: 10, l: 'Moneda'}, {v: 5, l: 'Moneda'}, {v: 2, l: 'Moneda'}, {v: 1, l: 'Moneda'}, {v: 0.5, l: 'Moneda'}

                        <span class="text-xs text-gray-400">x</span>

                        <input type="number" inputmode="decimal" pattern="[0-9]*" id="den_${i}" data-val="${d.v}" min="0" class="den-inp calc-trigger w-16 sm:w-20 text-center border border-gray-300 rounded py-2 text-base focus:ring-2 focus:ring-blue-500 outline-none bg-gray-50" placeholder="0">

                </div>

        // --- CÁLCULOS CENTRALES ---

        const fmt = n => new Intl.NumberFormat('es-MX', { style: 'currency', currency: 'MXN' }).format(n || 0);

        const getV = id => parseFloat(document.getElementById(id).value) || 0;


        function calculateAll() {

            // Tarjetas

            const tManana = getV('t_manana');

            const tTarde = getV('t_tarde');

            const subTickets = tManana + tTarde;

            const tTerminal = getV('t_terminal');

            const diffTarjetas = tTerminal - subTickets;

            

            document.getElementById('out_subtotal_tickets').textContent = fmt(subTickets);

            const elDT = document.getElementById('out_diff_tarjetas');

            elDT.textContent = fmt(diffTarjetas);

            elDT.className = `text-3xl font-black ${diffTarjetas === 0 ? 'text-green-600' : (diffTarjetas > 0 ? 'text-blue-600' : 'text-red-600')}`;


            // Sistema

            const sVentas = getV('s_ventas');

            const sEnt = getV('s_entradas');

            const sSal = getV('s_salidas');

            

            const totalConsumos = consumos.reduce((sum, c) => sum + (parseFloat(c.monto) || 0), 0);

            document.getElementById('out_total_consumos').textContent = fmt(totalConsumos);

            

            const totalAjustes = ajustes.reduce((sum, a) => sum + ((parseFloat(a.monto) || 0) * parseInt(a.signo)), 0);

            document.getElementById('out_total_ajustes').textContent = fmt(totalAjustes);


            const sComp = document.getElementById('s_compensar').checked ? (-diffTarjetas) : 0;


            const esperado = sVentas + sEnt - sSal - totalConsumos + totalAjustes + sComp;

            document.getElementById('out_efectivo_esperado').textContent = fmt(esperado);


            // Arqueo Físico

            let totalDesglose = 0;

            document.querySelectorAll('.den-inp').forEach(inp => {

                totalDesglose += (parseFloat(inp.dataset.val) * (parseInt(inp.value) || 0));

            });


            const usd = getV('a_usd_monto');

            const usdTc = getV('a_usd_rate');

            const usdMxn = usd * usdTc;

            document.getElementById('out_usd_mxn').textContent = fmt(usdMxn);

            

            const totalSobre = totalDesglose + usdMxn;

            document.getElementById('out_total_sobre').textContent = fmt(totalSobre);

            

            const fondo = getV('a_fondo_monto');

            const arqueoGeneral = totalSobre + fondo;


            // Auditoría

            const diffAud = arqueoGeneral - esperado;

            

            document.getElementById('aud_esperado').textContent = fmt(esperado);

            document.getElementById('aud_arqueo').textContent = fmt(arqueoGeneral);

            document.getElementById('aud_diff').textContent = fmt(diffAud);


            const bAud = document.getElementById('aud_box');

            const sAud = document.getElementById('aud_status');


            if(diffAud === 0) {

                bAud.className = 'p-6 flex flex-col items-center justify-center transition-colors bg-green-600';

                sAud.textContent = 'CUADRE EXACTO';

                sAud.className = 'mt-2 text-sm font-black uppercase bg-green-800 px-3 py-1 rounded tracking-widest text-green-100 shadow-inner';

            } else if (diffAud > 0) {

                bAud.className = 'p-6 flex flex-col items-center justify-center transition-colors bg-blue-600';

                sAud.textContent = 'SOBRANTE';

                sAud.className = 'mt-2 text-sm font-black uppercase bg-blue-800 px-3 py-1 rounded tracking-widest text-blue-100 shadow-inner';

            } else {

                bAud.className = 'p-6 flex flex-col items-center justify-center transition-colors bg-red-600';

                sAud.textContent = 'FALTANTE';

                sAud.className = 'mt-2 text-sm font-black uppercase bg-red-800 px-3 py-1 rounded tracking-widest text-red-100 shadow-inner';

            }


            return { tManana, tTarde, subTickets, tTerminal, diffTarjetas, sVentas, sEnt, sSal, totalConsumos, totalAjustes, sComp, esperado, usd, usdTc, usdMxn, totalSobre, fondo, arqueoGeneral, diffAud };

        }


        // Attach global events

        document.getElementById('app-container').addEventListener('input', e => {

            if(e.target.classList.contains('calc-trigger')) calculateAll();

        });

            document.getElementById('pdf_t_tarde').textContent = fmt(d.tTarde);

            document.getElementById('pdf_s_entradas').textContent = fmt(d.sEnt);

                    const c = v >= 0 ? 'text-green-600' : 'text-red-600';

                    const pre = v >= 0 ? '+' : '';

                    return `<div class="flex justify-between text-[11px] mb-1"><span>* ${a.motivo || 'Sin desc'}:</span> <span class="${c}">${pre}${fmt(v)}</span></div>`;

            document.getElementById('pdf_a_fondo').textContent = fmt(d.fondo);

                    const val = parseFloat(inp.dataset.val);

                    const l = denoms[idx].l;
